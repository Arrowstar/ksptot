  874       *Tp = 0.8 * (muDoubleScalarPower(q, 5.0) - 1.0);
  875   
  876       /*  T''(x) */
  877       st.site = &kd_emlrtRSI;
  878       b_st.site = &nb_emlrtRSI;
  879       *Tpp = *Tp + 1.7142857142857142 * (1.0 - muDoubleScalarPower(q, 7.0));
  880   
  881       /*  T'''(x) */
  882       st.site = &ld_emlrtRSI;
  883       b_st.site = &nb_emlrtRSI;
  884     } else if (muDoubleScalarAbs(x - 1.0) < 0.01) {
  885       /*  near-parabolic; compute with series */
  886       /*  evaluate sigma */
  887       st.site = &md_emlrtRSI;
  888       sigmax(-E, &y, &dsigdx1, &d2sigdx21, &f);
  889       st.site = &nd_emlrtRSI;
  890       sigmax(-E * q * q, &f, &z, &d2sigdx22, &g);
  891   
  892       /*  T(x) */
  893       st.site = &od_emlrtRSI;
  894       b_st.site = &nb_emlrtRSI;
  895       *T = y - muDoubleScalarPower(q, 3.0) * f;
  896   
  897       /*  T'(x) */
  898       st.site = &pd_emlrtRSI;
  899       b_st.site = &nb_emlrtRSI;
  900       *Tp = 2.0 * x * (muDoubleScalarPower(q, 5.0) * z - dsigdx1);
  901   
  902       /*  T''(x) */
  903       st.site = &qd_emlrtRSI;
  904       b_st.site = &nb_emlrtRSI;
  905       st.site = &qd_emlrtRSI;
  906       b_st.site = &nb_emlrtRSI;
  907       *Tpp = *Tp / x + 4.0 * (x * x) * (d2sigdx21 - muDoubleScalarPower(q, 7.0) *
  908         d2sigdx22);
  909   
  910       /*  T'''(x) */
  911       st.site = &rd_emlrtRSI;
  912       b_st.site = &nb_emlrtRSI;
  913     } else {
  914       /*  all other cases */
  915       /*  compute all substitution functions */
  916       y = muDoubleScalarAbs(E);
  917       st.site = &sd_emlrtRSI;
  918       b_sqrt(&st, &y);
  919       st.site = &td_emlrtRSI;
  920       b_st.site = &nb_emlrtRSI;
  921       st.site = &td_emlrtRSI;
  922       f = 1.0 + q * q * E;
  923       if (f < 0.0) {
  924         b_st.site = &lb_emlrtRSI;
  925         c_st.site = &lb_emlrtRSI;
  926         error(&c_st);
  927       }
  928   
  929       z = muDoubleScalarSqrt(f);
  930       f = y * (z - q * x);
  931       g = x * z - q * E;
  932   
  933       /*  BUGFIX: (Simon Tardivel) this line is incorrect for E==0 and f+g==0 */
  934       /*  d  = (E < 0)*(atan2(f, g) + pi*m) + (E > 0)*log( max(0, f + g) ); */
  935       /*  it should be written out like so: */
  936       if (E < 0.0) {
  937         f = muDoubleScalarAtan2(f, g) + 3.1415926535897931 * m;
  938       } else if (E == 0.0) {
  939         f = 0.0;
  940       } else {
  941         st.site = &ud_emlrtRSI;
  942         f = muDoubleScalarLog(muDoubleScalarMax(0.0, f + g));
  943       }
  944   
  945       /*  T(x) */
  946       *T = 2.0 * ((x - q * z) - f / y) / E;
  947   
  948       /*   T'(x) */
  949       st.site = &vd_emlrtRSI;
  950       b_st.site = &nb_emlrtRSI;
  951       *Tp = ((4.0 - 4.0 * muDoubleScalarPower(q, 3.0) * x / z) - 3.0 * x * *T) / E;
  952   
  953       /*  T''(x) */
  954       st.site = &wd_emlrtRSI;
  955       b_st.site = &nb_emlrtRSI;
  956       st.site = &wd_emlrtRSI;
  957       b_st.site = &nb_emlrtRSI;
  958       st.site = &wd_emlrtRSI;
  959       b_st.site = &nb_emlrtRSI;
  960       st.site = &wd_emlrtRSI;
  961       b_st.site = &nb_emlrtRSI;
  962       *Tpp = ((-4.0 * muDoubleScalarPower(q, 3.0) / z * (1.0 - q * q * (x * x) /
  963                 (z * z)) - 3.0 * *T) - 3.0 * x * *Tp) / E;
  964   
  965       /*  T'''(x) */
  966       st.site = &xd_emlrtRSI;
  967       b_st.site = &nb_emlrtRSI;
  968       st.site = &xd_emlrtRSI;
  969       b_st.site = &nb_emlrtRSI;
  970       st.site = &xd_emlrtRSI;
  971       b_st.site = &nb_emlrtRSI;
  972       st.site = &xd_emlrtRSI;
  973       b_st.site = &nb_emlrtRSI;
  974       st.site = &xd_emlrtRSI;
  975       b_st.site = &nb_emlrtRSI;
  976       st.site = &xd_emlrtRSI;
  977       b_st.site = &nb_emlrtRSI;
  978       st.site = &xd_emlrtRSI;
  979       b_st.site = &nb_emlrtRSI;
  980     }
  981   }

  984     [3], const real_T r2vec[3], real_T tf, real_T m, real_T muC, real_T V1[3],
  985     real_T V2[3], real_T extremal_distances[2], real_T *exitflag)

  987     real_T r1;
  988     int32_T iterations;
  989     real_T r2;

  993     real_T r2unit[3];

  999     real_T s;

 1028     e_st.tls = d_st.tls;

 1032     /*  Very reliable, moderately fast for both simple and complicated cases */
 1033     /*  ----------------------------------------------------------------- */
 1034     /* #coder */

 1036     /* LAMBERT_LANCASTERBLANCHARD       High-Thrust Lambert-targeter */
 1037     /*  */

 1039     /* Lancaster & Blancard, as described in their 1969 paper. Initial values, */

 1042     /* } */

 1044     /*  */

 1051     /*  manipulate input */
 1052     /*  optimum for numerical noise v.s. actual precision */
 1053     r1 = 0.0;
 1054     for (iterations = 0; iterations < 3; iterations++) {
 1055       r1 += r1vec[iterations] * r1vec[iterations];
 1056     }
 1057   
 1058     st.site = &xb_emlrtRSI;
 1059     b_sqrt(&st, &r1);
 1060   
 1061     /*  magnitude of r1vec */
 1062     r2 = 0.0;
 1063     for (iterations = 0; iterations < 3; iterations++) {
 1064       r2 += r2vec[iterations] * r2vec[iterations];
 1065     }
 1066   
 1067     st.site = &yb_emlrtRSI;
 1068     b_sqrt(&st, &r2);
 1069   
 1070     /*  magnitude of r2vec */
 1071     /*  unit vector of r1vec */
 1072     /*  unit vector of r2vec */
 1073     cross(r1vec, r2vec, crsprod);
 1074   
 1075     /*  cross product of r1vec and r2vec */
 1076     mcrsprd = 0.0;
 1077     for (iterations = 0; iterations < 3; iterations++) {
 1078       r1unit[iterations] = r1vec[iterations] / r1;
 1079       r2unit[iterations] = r2vec[iterations] / r2;
 1080       mcrsprd += crsprod[iterations] * crsprod[iterations];
 1081     }
 1082   
 1083     st.site = &ac_emlrtRSI;
 1084     b_sqrt(&st, &mcrsprd);
 1085   
 1086     /*  magnitude of that cross product */
 1087     for (iterations = 0; iterations < 3; iterations++) {
 1088       b_crsprod[iterations] = crsprod[iterations] / mcrsprd;
 1089     }
 1090   
 1091     cross(b_crsprod, r1unit, th1unit);
 1092   
 1093     /*  unit vectors in the tangential-directions */
 1094     for (iterations = 0; iterations < 3; iterations++) {
 1095       b_crsprod[iterations] = crsprod[iterations] / mcrsprd;
 1096     }
 1097   
 1098     cross(b_crsprod, r2unit, crsprod);
 1099   
 1100     /*  make 100.4% sure it's in (-1 <= x <= +1) */
 1101     mcrsprd = 0.0;
 1102     for (iterations = 0; iterations < 3; iterations++) {
 1103       mcrsprd += r1vec[iterations] * r2vec[iterations];
 1104     }
 1105   
 1106     dth = muDoubleScalarMax(-1.0, muDoubleScalarMin(1.0, mcrsprd / r1 / r2));
 1107     st.site = &bc_emlrtRSI;
 1108     b_acos(&st, &dth);
 1109   
 1110     /*  turn angle */
 1111     /*  if the long way was selected, the turn-angle must be negative */
 1112     /*  to take care of the direction of final velocity */
 1113     mcrsprd = muDoubleScalarSign(tf);
 1114     tf = muDoubleScalarAbs(tf);
 1115     if (mcrsprd < 0.0) {
 1116       dth -= 6.2831853071795862;
 1117     }
 1118   
 1119     /*  left-branch */
 1120     leftbranch = muDoubleScalarSign(m);
 1121     m = muDoubleScalarAbs(m);
 1122   
 1123     /*  define constants */
 1124     st.site = &cc_emlrtRSI;
 1125     b_st.site = &nb_emlrtRSI;
 1126     c_st.site = &ob_emlrtRSI;
 1127     st.site = &cc_emlrtRSI;
 1128     b_st.site = &nb_emlrtRSI;
 1129     c_st.site = &ob_emlrtRSI;
 1130     c = (r1 * r1 + r2 * r2) - 2.0 * r1 * r2 * muDoubleScalarCos(dth);
 1131     st.site = &cc_emlrtRSI;
 1132     b_sqrt(&st, &c);
 1133     s = ((r1 + r2) + c) / 2.0;
 1134     st.site = &dc_emlrtRSI;
 1135     b_st.site = &nb_emlrtRSI;
 1136     c_st.site = &ob_emlrtRSI;
 1137     Vr1 = 8.0 * muC / muDoubleScalarPower(s, 3.0);
 1138     st.site = &dc_emlrtRSI;
 1139     b_sqrt(&st, &Vr1);
 1140     T = Vr1 * tf;
 1141     TmTM = r1 * r2;
 1142     st.site = &ec_emlrtRSI;
 1143     b_sqrt(&st, &TmTM);
 1144     q = TmTM / s * muDoubleScalarCos(dth / 2.0);
 1145   
 1146     /*  general formulae for the initial values (Gooding) */
 1147     /*  ------------------------------------------------- */
 1148     /*  some initial values */
 1149     st.site = &fc_emlrtRSI;
 1150     T0 = LancasterBlanchard(&st, 0.0, q, m);
 1151     Td = T0 - T;
 1152     st.site = &gc_emlrtRSI;
 1153     b_st.site = &nb_emlrtRSI;
 1154     c_st.site = &ob_emlrtRSI;
 1155     phr = b_mod(2.0 * muDoubleScalarAtan2(1.0 - q * q, 2.0 * q));
 1156   
 1157     /*  initial output is pessimistic */
 1158     for (iterations = 0; iterations < 3; iterations++) {
 1159       V1[iterations] = rtNaN;
 1160       V2[iterations] = rtNaN;
 1161     }
 1162   
 1163     for (iterations = 0; iterations < 2; iterations++) {
 1164       extremal_distances[iterations] = rtNaN;
 1165     }
 1166   
 1167     /*  single-revolution case */
 1168     guard1 = false;
 1169     if (m == 0.0) {
 1170       if (Td > 0.0) {
 1171         phr = T0 * Td / 4.0 / T;
 1172       } else {
 1173         xM0 = Td / (4.0 - Td);
 1174         Vr1 = -Td / (T + T0 / 2.0);
 1175         st.site = &hc_emlrtRSI;
 1176         b_sqrt(&st, &Vr1);
 1177         mcrsprd = 2.0 - phr / 3.1415926535897931;
 1178         st.site = &ic_emlrtRSI;
 1179         b_sqrt(&st, &mcrsprd);
 1180         mcrsprd = xM0 + 1.7 * mcrsprd;
 1181         if (mcrsprd >= 0.0) {
 1182           TmTM = xM0;
 1183         } else {
 1184           st.site = &jc_emlrtRSI;
 1185           b_st.site = &ob_emlrtRSI;
 1186           TmTM = xM0 + muDoubleScalarPower(-mcrsprd, 0.0625) * (-Vr1 - xM0);
 1187         }
 1188   
 1189         st.site = &kc_emlrtRSI;
 1190         b_st.site = &nb_emlrtRSI;
 1191         c_st.site = &ob_emlrtRSI;
 1192         Vr1 = 1.0 + xM0;
 1193         st.site = &kc_emlrtRSI;
 1194         b_sqrt(&st, &Vr1);
 1195         phr = ((1.0 + TmTM * (1.0 + xM0) / 2.0) - 0.03 * (TmTM * TmTM) * Vr1) *
 1196           TmTM;
 1197       }
 1198   
 1199       /*  this estimate might not give a solution */
 1200       if (phr < -1.0) {
 1201         *exitflag = -1.0;
 1202       } else {
 1203         /*  multi-revolution case */
 1204         guard1 = true;
 1205       }
 1206     } else {
 1207       /*  determine minimum Tp(x) */
 1208       mcrsprd = 4.0 / (9.42477796076938 * (2.0 * m + 1.0));
 1209       if (phr < 3.1415926535897931) {
 1210         TmTM = phr / 3.1415926535897931;
 1211         st.site = &lc_emlrtRSI;
 1212         b_st.site = &nb_emlrtRSI;
 1213         c_st.site = &ob_emlrtRSI;
 1214         if (TmTM < 0.0) {
 1215           d_st.site = &pb_emlrtRSI;
 1216           e_st.site = &pb_emlrtRSI;
 1217           c_error(&e_st);
 1218         }
 1219   
 1220         xM0 = mcrsprd * muDoubleScalarPower(TmTM, 0.125);
 1221       } else if (phr > 3.1415926535897931) {
 1222         st.site = &mc_emlrtRSI;
 1223         TmTM = 2.0 - phr / 3.1415926535897931;
 1224         b_st.site = &nb_emlrtRSI;
 1225         c_st.site = &ob_emlrtRSI;
 1226         if (TmTM < 0.0) {
 1227           d_st.site = &pb_emlrtRSI;
 1228           e_st.site = &pb_emlrtRSI;
 1229           c_error(&e_st);
 1230         }
 1231   
 1232         xM0 = mcrsprd * (2.0 - muDoubleScalarPower(TmTM, 0.125));
 1233   
 1234         /*  EMLMEX requires this one */
 1235       } else {
 1236         xM0 = 0.0;
 1237       }
 1238   
 1239       /*  use Halley's method */
 1240       Tp = rtInf;
 1241       iterations = 0;
 1242       do {
 1243         exitg2 = 0;
 1244         if (muDoubleScalarAbs(Tp) > 1.0E-12) {
 1245           /*  iterations */
 1246           iterations++;
 1247   
 1248           /*  compute first three derivatives */
 1249           st.site = &nc_emlrtRSI;
 1250           b_LancasterBlanchard(&st, xM0, q, m, &Vr1, &Tp, &b_gamma, &mcrsprd);
 1251   
 1252           /*  new value of xM */
 1253           xMp = xM0;
 1254           st.site = &oc_emlrtRSI;
 1255           b_st.site = &ob_emlrtRSI;
 1256           xM0 -= 2.0 * Tp * b_gamma / (2.0 * (b_gamma * b_gamma) - Tp * mcrsprd);
 1257   
 1258           /*  escape clause */
 1259           Vr1 = c_mod(iterations);
 1260           st.site = &pc_emlrtRSI;
 1261           toLogicalCheck(&st, Vr1);
 1262           if (Vr1 != 0.0) {
 1263             xM0 = (xMp + xM0) / 2.0;
 1264           }
 1265   
 1266           /*  the method might fail. Exit in that case */
 1267           if (*emlrtBreakCheckR2012bFlagVar != 0) {
 1268             emlrtBreakCheckR2012b(sp);
 1269           }
 1270   
 1271           if (iterations > 25) {
 1272             *exitflag = -2.0;
 1273             exitg2 = 1;
 1274           }
 1275         } else {
 1276           /*  xM should be elliptic (-1 < x < 1) */
 1277           /*  (this should be impossible to go wrong) */
 1278           exitg2 = 2;
 1279         }
 1280       } while (exitg2 == 0);
 1281   
 1282       if (exitg2 == 1) {
 1283       } else if ((xM0 < -1.0) || (xM0 > 1.0)) {
 1284         *exitflag = -1.0;
 1285       } else {
 1286         /*  corresponding time */
 1287         st.site = &qc_emlrtRSI;
 1288         mcrsprd = LancasterBlanchard(&st, xM0, q, m);
 1289   
 1290         /*  T should lie above the minimum T */
 1291         if (mcrsprd > T) {
 1292           *exitflag = -1.0;
 1293         } else {
 1294           /*  find two initial values for second solution (again with lambda-type patch) */
 1295           /*  -------------------------------------------------------------------------- */
 1296           /*  some initial values */
 1297           TmTM = T - mcrsprd;
 1298           st.site = &rc_emlrtRSI;
 1299           c_LancasterBlanchard(&st, xM0, q, m, &Vr1, &Tp, &b_gamma);
 1300   
 1301           /*  first estimate (only if m > 0) */
 1302           if (leftbranch > 0.0) {
 1303             st.site = &sc_emlrtRSI;
 1304             b_st.site = &nb_emlrtRSI;
 1305             c_st.site = &ob_emlrtRSI;
 1306             phr = TmTM / (b_gamma / 2.0 + TmTM / ((1.0 - xM0) * (1.0 - xM0)));
 1307             st.site = &sc_emlrtRSI;
 1308             b_sqrt(&st, &phr);
 1309             mcrsprd = xM0 + phr;
 1310             st.site = &tc_emlrtRSI;
 1311             b_st.site = &nb_emlrtRSI;
 1312             c_st.site = &ob_emlrtRSI;
 1313             mcrsprd = 4.0 * mcrsprd / (4.0 + TmTM) + (1.0 - mcrsprd) * (1.0 -
 1314               mcrsprd);
 1315             Vr1 = mcrsprd;
 1316             st.site = &uc_emlrtRSI;
 1317             b_sqrt(&st, &Vr1);
 1318             phr = phr * (1.0 - ((1.0 + m) + (dth - 0.5)) / (1.0 + 0.15 * m) * phr *
 1319                          (mcrsprd / 2.0 + 0.03 * phr * Vr1)) + xM0;
 1320   
 1321             /*  first estimate might not be able to yield possible solution */
 1322             if (phr > 1.0) {
 1323               *exitflag = -1.0;
 1324             } else {
 1325               /*  second estimate (only if m > 0) */
 1326               guard1 = true;
 1327             }
 1328           } else {
 1329             if (Td > 0.0) {
 1330               st.site = &vc_emlrtRSI;
 1331               b_st.site = &nb_emlrtRSI;
 1332               c_st.site = &ob_emlrtRSI;
 1333               Vr1 = mcrsprd / (b_gamma / 2.0 - TmTM * (b_gamma / 2.0 / (T0 -
 1334                 mcrsprd) - 1.0 / (xM0 * xM0)));
 1335               st.site = &vc_emlrtRSI;
 1336               b_sqrt(&st, &Vr1);
 1337               phr = xM0 - Vr1;
 1338             } else {
 1339               TmTM = Td / (4.0 - Td);
 1340               Vr1 = 2.0 * (1.0 - phr);
 1341               st.site = &wc_emlrtRSI;
 1342               b_sqrt(&st, &Vr1);
 1343               mcrsprd = TmTM + 1.7 * Vr1;
 1344               if (!(mcrsprd >= 0.0)) {
 1345                 st.site = &xc_emlrtRSI;
 1346                 b_st.site = &nb_emlrtRSI;
 1347                 c_st.site = &ob_emlrtRSI;
 1348                 Vr1 = muDoubleScalarPower(-mcrsprd, 0.125);
 1349                 st.site = &xc_emlrtRSI;
 1350                 b_sqrt(&st, &Vr1);
 1351                 mcrsprd = -Td / (1.5 * T0 - Td);
 1352                 st.site = &xc_emlrtRSI;
 1353                 b_sqrt(&st, &mcrsprd);
 1354                 TmTM -= Vr1 * (TmTM + mcrsprd);
 1355               }
 1356   
 1357               mcrsprd = 4.0 / (4.0 - Td);
 1358               Vr1 = mcrsprd;
 1359               st.site = &yc_emlrtRSI;
 1360               b_sqrt(&st, &Vr1);
 1361               phr = TmTM * (1.0 + ((1.0 + m) + 0.24 * (dth - 0.5)) / (1.0 + 0.15 *
 1362                 m) * TmTM * (mcrsprd / 2.0 - 0.03 * TmTM * Vr1));
 1363             }
 1364   
 1365             /*  estimate might not give solutions */
 1366             if (phr < -1.0) {
 1367               *exitflag = -1.0;
 1368             } else {
 1369               guard1 = true;
 1370             }
 1371           }
 1372         }
 1373       }
 1374     }
 1375   
 1376     if (guard1) {
 1377       /*  find root of Lancaster & Blancard's function */
 1378       /*  -------------------------------------------- */
 1379       /*  (Halley's method) */
 1380       mcrsprd = rtInf;
 1381       iterations = 0;
 1382       do {
 1383         exitg1 = 0;
 1384         if (muDoubleScalarAbs(mcrsprd) > 1.0E-12) {
 1385           /*  iterations */
 1386           iterations++;
 1387   
 1388           /*  compute function value, and first two derivatives */
 1389           st.site = &ad_emlrtRSI;
 1390           c_LancasterBlanchard(&st, phr, q, m, &mcrsprd, &Tp, &b_gamma);
 1391   
 1392           /*  find the root of the *difference* between the */
 1393           /*  function value [T_x] and the required time [T] */
 1394           mcrsprd -= T;
 1395   
 1396           /*  new value of x */
 1397           TmTM = phr;
 1398           st.site = &bd_emlrtRSI;
 1399           b_st.site = &nb_emlrtRSI;
 1400           c_st.site = &ob_emlrtRSI;
 1401           phr -= 2.0 * mcrsprd * Tp / (2.0 * (Tp * Tp) - mcrsprd * b_gamma);
 1402   
 1403           /*  escape clause */
 1404           Vr1 = c_mod(iterations);
 1405           st.site = &cd_emlrtRSI;
 1406           toLogicalCheck(&st, Vr1);
 1407           if (Vr1 != 0.0) {
 1408             phr = (TmTM + phr) / 2.0;
 1409           }
 1410   
 1411           /*  Halley's method might fail */
 1412           if (*emlrtBreakCheckR2012bFlagVar != 0) {
 1413             emlrtBreakCheckR2012b(sp);
 1414           }
 1415   
 1416           if (iterations > 25) {
 1417             *exitflag = -2.0;
 1418             exitg1 = 1;
 1419           }
 1420         } else {
 1421           /*  calculate terminal velocities */
 1422           /*  ----------------------------- */
 1423           /*  constants required for this calculation */
 1424           b_gamma = muC * s / 2.0;
 1425           st.site = &dd_emlrtRSI;
 1426           b_sqrt(&st, &b_gamma);
 1427           if (c == 0.0) {
 1428             leftbranch = 1.0;
 1429             mcrsprd = 0.0;
 1430             TmTM = muDoubleScalarAbs(phr);
 1431           } else {
 1432             st.site = &ed_emlrtRSI;
 1433             b_st.site = &nb_emlrtRSI;
 1434             c_st.site = &ob_emlrtRSI;
 1435             TmTM = r1 * r2 / (c * c);
 1436             st.site = &ed_emlrtRSI;
 1437             if (TmTM < 0.0) {
 1438               b_st.site = &lb_emlrtRSI;
 1439               c_st.site = &lb_emlrtRSI;
 1440               error(&c_st);
 1441             }
 1442   
 1443             leftbranch = 2.0 * muDoubleScalarSqrt(TmTM) * muDoubleScalarSin(dth /
 1444               2.0);
 1445             mcrsprd = (r1 - r2) / c;
 1446             st.site = &fd_emlrtRSI;
 1447             b_st.site = &nb_emlrtRSI;
 1448             c_st.site = &ob_emlrtRSI;
 1449             st.site = &fd_emlrtRSI;
 1450             b_st.site = &nb_emlrtRSI;
 1451             c_st.site = &ob_emlrtRSI;
 1452             st.site = &fd_emlrtRSI;
 1453             TmTM = 1.0 + q * q * (phr * phr - 1.0);
 1454             if (TmTM < 0.0) {
 1455               b_st.site = &lb_emlrtRSI;
 1456               c_st.site = &lb_emlrtRSI;
 1457               error(&c_st);
 1458             }
 1459   
 1460             TmTM = muDoubleScalarSqrt(TmTM);
 1461           }
 1462   
 1463           /*  radial component */
 1464           Vr1 = b_gamma * ((q * TmTM - phr) - mcrsprd * (q * TmTM + phr)) / r1;
 1465           xM0 = -b_gamma * ((q * TmTM - phr) + mcrsprd * (q * TmTM + phr)) / r2;
 1466   
 1467           /*  tangential component */
 1468           xMp = leftbranch * b_gamma * (TmTM + q * phr) / r1;
 1469           mcrsprd = leftbranch * b_gamma * (TmTM + q * phr) / r2;
 1470   
 1471           /*  Cartesian velocity */
 1472           for (iterations = 0; iterations < 3; iterations++) {
 1473             V1[iterations] = xMp * th1unit[iterations] + Vr1 * r1unit[iterations];
 1474             V2[iterations] = mcrsprd * crsprod[iterations] + xM0 *
 1475               r2unit[iterations];
 1476           }
 1477   
 1478           /*  exitflag */
 1479           *exitflag = 1.0;
 1480   
 1481           /*  (success) */
 1482           /*  also determine minimum/maximum distance */
 1483           st.site = &gd_emlrtRSI;
 1484           b_st.site = &nb_emlrtRSI;
 1485           c_st.site = &ob_emlrtRSI;
 1486   
 1487           /*  semi-major axis */
 1488           st.site = &hd_emlrtRSI;
 1489           minmax_distances(&st, r1vec, r1, r1vec, r2, dth, s / 2.0 / (1.0 - phr *
 1490             phr), V1, V2, m, muC, extremal_distances);
 1491           exitg1 = 1;
 1492         }
 1493       } while (exitg1 == 0);
 1494     }
 1495   }

 1497   static void minmax_distances(const emlrtStack *sp, const real_T r1vec[3], real_T
 1498     r1, const real_T r2vec[3], real_T r2, real_T dth, real_T a, const real_T V1[3],
 1499     const real_T V2[3], real_T m, real_T muC, real_T extremal_distances[2])
 1500   {
 1501     real_T minimum_distance;
 1502     real_T maximum_distance;
 1503     real_T y;
 1504     real_T theta1;
 1505     int32_T k;
 1506     real_T absxk;
 1507     real_T theta2;
 1508     real_T evec[3];
 1509     real_T e;
 1510     real_T pericenter;
 1511     real_T apocenter;
 1512     real_T b_y;
 1513     real_T c_y;
 1514     real_T d_y;
 1515     int32_T exponent;
 1516     emlrtStack st;
 1517     emlrtStack b_st;
 1518     emlrtStack c_st;
 1519     st.prev = sp;
 1520     st.tls = sp->tls;
 1521     b_st.prev = &st;
 1522     b_st.tls = st.tls;
 1523     c_st.prev = &b_st;
 1524     c_st.tls = b_st.tls;
 1525   
 1526     /*  ----------------------------------------------------------------- */
 1527     /*  Helper functions */
 1528     /*  ----------------------------------------------------------------- */
 1529     /*  compute minimum and maximum distances to the central body */
 1530     /*  default - minimum/maximum of r1,r2 */
 1531     minimum_distance = muDoubleScalarMin(r1, r2);
 1532     maximum_distance = muDoubleScalarMax(r1, r2);
 1533   
 1534     /*  was the longway used or not? */
 1535     /*  eccentricity vector (use triple product identity) */
 1536     y = 0.0;
 1537     theta1 = 0.0;
 1538     for (k = 0; k < 3; k++) {
 1539       y += V1[k] * V1[k];
 1540       theta1 += V1[k] * r1vec[k];
 1541     }
 1542   
 1543     /*  eccentricity */
 1544     absxk = 0.0;
 1545     for (k = 0; k < 3; k++) {
 1546       theta2 = (y * r1vec[k] - theta1 * V1[k]) / muC - r1vec[k] / r1;
 1547       absxk += theta2 * theta2;
 1548       evec[k] = theta2;
 1549     }
 1550   
 1551     st.site = &be_emlrtRSI;
 1552     if (absxk < 0.0) {
 1553       b_st.site = &lb_emlrtRSI;
 1554       c_st.site = &lb_emlrtRSI;
 1555       error(&c_st);
 1556     }
 1557   
 1558     e = muDoubleScalarSqrt(absxk);
 1559   
 1560     /*  apses */
 1561     pericenter = a * (1.0 - e);
 1562     apocenter = rtInf;
 1563   
 1564     /*  parabolic/hyperbolic case */
 1565     if (e < 1.0) {
 1566       apocenter = a * (1.0 + e);
 1567     }
 1568   
 1569     /*  elliptic case */
 1570     /*  since we have the eccentricity vector, we know exactly where the */
 1571     /*  pericenter lies. Use this fact, and the given value of [dth], to */
 1572     /*  cross-check if the trajectory goes past it */
 1573     if (m > 0.0) {
 1574       /*  obvious case (always elliptical and both apses are traversed) */
 1575       minimum_distance = pericenter;
 1576       maximum_distance = apocenter;
 1577     } else {
 1578       /*  less obvious case */
 1579       /*  compute theta1&2 ( use (AxB)-(CxD) = (CÂ·B)(DÂ·A) - (CÂ·A)(BÂ·D) )) */
 1580       y = 0.0;
 1581       theta1 = 0.0;
 1582       theta2 = 0.0;
 1583       b_y = 0.0;
 1584       c_y = 0.0;
 1585       d_y = 0.0;
 1586   
 1587       /*  make 100.4% sure it's in (-1 <= theta12 <= +1) */
 1588       absxk = 0.0;
 1589       for (k = 0; k < 3; k++) {
 1590         y += evec[k] * V1[k];
 1591         theta1 += r1vec[k] * evec[k];
 1592         theta2 += r1vec[k] * V1[k];
 1593         b_y += evec[k] * V2[k];
 1594         c_y += r2vec[k] * evec[k];
 1595         d_y += r2vec[k] * V2[k];
 1596         absxk += r1vec[k] / r1 * (evec[k] / e);
 1597       }
 1598   
 1599       absxk = muDoubleScalarMax(-1.0, muDoubleScalarMin(1.0, absxk));
 1600       st.site = &ce_emlrtRSI;
 1601       b_acos(&st, &absxk);
 1602       theta1 = muDoubleScalarSign(r1 * r1 * y - theta1 * theta2) * absxk;
 1603       y = 0.0;
 1604       for (k = 0; k < 3; k++) {
 1605         y += r2vec[k] / r2 * (evec[k] / e);
 1606       }
 1607   
 1608       absxk = muDoubleScalarMax(-1.0, muDoubleScalarMin(1.0, y));
 1609       st.site = &de_emlrtRSI;
 1610       b_acos(&st, &absxk);
 1611       theta2 = muDoubleScalarSign(r2 * r2 * b_y - c_y * d_y) * absxk;
 1612   
 1613       /*  points 1&2 are on opposite sides of the symmetry axis -- minimum */
 1614       /*  and maximum distance depends both on the value of [dth], and both */
 1615       /*  [theta1] and [theta2] */
 1616       if (theta1 * theta2 < 0.0) {
 1617         /*  if |th1| + |th2| = turnangle, we know that the pericenter was */
 1618         /*  passed */
 1619         st.site = &ee_emlrtRSI;
 1620         absxk = muDoubleScalarAbs(dth);
 1621         if ((!muDoubleScalarIsInf(absxk)) && (!muDoubleScalarIsNaN(absxk))) {
 1622           if (absxk <= 2.2250738585072014E-308) {
 1623             absxk = 4.94065645841247E-324;
 1624           } else {
 1625             frexp(absxk, &exponent);
 1626             absxk = ldexp(1.0, exponent - 53);
 1627           }
 1628         } else {
 1629           absxk = rtNaN;
 1630         }
 1631   
 1632         if (muDoubleScalarAbs((muDoubleScalarAbs(theta1) + muDoubleScalarAbs
 1633                                (theta2)) - dth) < 5.0 * absxk) {
 1634           minimum_distance = pericenter;
 1635   
 1636           /*  this condition can only be false for elliptic cases, and */
 1637           /*  when it is indeed false, we know that the orbit passed */
 1638           /*  apocenter */
 1639         } else {
 1640           maximum_distance = apocenter;
 1641         }
 1642   
 1643         /*  points 1&2 are on the same side of the symmetry axis. Only if the */
 1644         /*  long-way was used are the min. and max. distances different from */
 1645         /*  the min. and max. values of the radii (namely, equal to the apses) */
 1646       } else {
 1647         if (muDoubleScalarAbs(dth) > 3.1415926535897931) {
 1648           minimum_distance = pericenter;
 1649           if (e < 1.0) {
 1650             maximum_distance = apocenter;
 1651           }
 1652         }
 1653       }
 1654     }
 1655   
 1656     /*  output argument */
 1657     extremal_distances[0] = minimum_distance;
 1658     extremal_distances[1] = maximum_distance;
 1659   }

 1662                      real_T *d3sigdx3)
 1663   {
 1664     int32_T k;
 1665     real_T powers[25];
 1666     real_T b_y;
 1667     real_T dv1[25];
 1668     static const int16_T iv0[25] = { 0, 2, 6, 12, 20, 30, 42, 56, 72, 90, 110, 132,
 1669       156, 182, 210, 240, 272, 306, 342, 380, 420, 462, 506, 552, 600 };
 1670

 1679     /*  powers of y */

 1699

 1701     dv1[0] = 1.0 / y;
 1702     dv1[1] = 1.0;
 1703     memcpy(&dv1[2], &powers[0], 23U * sizeof(real_T));

 1710     dv1[0] = 1.0 / y / y;
 1711     dv1[1] = 1.0 / y;
 1712     dv1[2] = 1.0;
 1713     memcpy(&dv1[3], &powers[0], 22U * sizeof(real_T));
 1714     *d3sigdx3 = 0.0;
 1715     for (k = 0; k < 25; k++) {
 1716       *d3sigdx3 += (real_T)iv1[k] * dv1[k] * an[k];
 1717     }
 1718   }
