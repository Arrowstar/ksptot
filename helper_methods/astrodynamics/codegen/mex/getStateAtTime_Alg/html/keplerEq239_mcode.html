1function E = keplerEq(M,e,eps)
 2% Function solves Kepler's equation M = E-e*sin(E)
 3% Input - Mean anomaly M [rad] , Eccentricity e and Epsilon 
 4% Output  eccentric anomaly E [rad]. 
 5   	En  = M;
 6	Ens = En - (En-e*sin(En)- M)/(1 - e*cos(En));
 7	while ( abs(Ens-En) > eps )
 8		En = Ens;
 9		Ens = En - (En - e*sin(En) - M)/(1 - e*cos(En));
10    end
11	E = Ens;
