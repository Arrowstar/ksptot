1function c = multiprod(a, b, idA, idB)
  2%MULTIPROD  Multiplying 1-D or 2-D subarrays contained in two N-D arrays.
  3%   C = MULTIPROD(A,B) is equivalent  to C = MULTIPROD(A,B,[1 2],[1 2])
  4%   C = MULTIPROD(A,B,[D1 D2]) is eq. to C = MULTIPROD(A,B,[D1 D2],[D1 D2])
  5%   C = MULTIPROD(A,B,D1) is equival. to C = MULTIPROD(A,B,D1,D1)
  6%
  7%   MULTIPROD performs multiple matrix products, with array expansion (AX)
  8%   enabled. Its first two arguments A and B are "block arrays" of any
  9%   size, containing one or more 1-D or 2-D subarrays, called "blocks" (*).
 10%   For instance, a 5×6×3 array may be viewed as an array containing five
 11%   6×3 blocks. In this case, its size is denoted by 5×(6×3). The 1 or 2
 12%   adjacent dimensions along which the blocks are contained are called the
 13%   "internal dimensions" (IDs) of the array (°).
 14%
 15%   1) 2-D by 2-D BLOCK(S) (*)
 16%         C = MULTIPROD(A, B, [DA1 DA2], [DB1 DB2]) contains the products
 17%         of the P×Q matrices in A by the R×S matrices in B. [DA1 DA2] are
 18%         the IDs of A; [DB1 DB2] are the IDs of B.
 19%
 20%   2) 2-D by 1-D BLOCK(S) (*)
 21%         C = MULTIPROD(A, B, [DA1 DA2], DB1) contains the products of the
 22%         P×Q matrices in A by the R-element vectors in B. The latter are
 23%         considered to be R×1 matrices. [DA1 DA2] are the IDs of A; DB1 is
 24%         the ID of B.
 25%
 26%   3) 1-D by 2-D BLOCK(S) (*)
 27%         C = MULTIPROD(A, B, DA1, [DB1 DB2]) contains the products of the 
 28%         Q-element vectors in A by the R×S matrices in B. The vectors in A
 29%         are considered to be 1×Q matrices. DA1 is the ID of A; [DB1 DB2]
 30%         are the IDs of B.
 31%
 32%   4) 1-D BY 1-D BLOCK(S) (*)
 33%      (a) If either SIZE(A, DA1) == 1 or SIZE(B, DB1) == 1, or both,
 34%             C = MULTIPROD(A, B, DA1, DB1) returns products of scalars by 
 35%             vectors, or vectors by scalars or scalars by scalars.
 36%      (b) If SIZE(A, DA1) == SIZE(B, DB1), 
 37%             C = MULTIPROD(A, B, [0 DA1], [DB1 0]) or 
 38%             C = MULTIPROD(A, B, DA1, DB1) virtually turns the vectors
 39%             contained in A and B into 1×P and P×1 matrices, respectively,
 40%             then returns their products, similar to scalar products.
 41%             Namely, C = DOT2(A, B, DA1, DB1) is equivalent to 
 42%             C = MULTIPROD(CONJ(A), B, [0 DA1], [DB1 0]).
 43%      (c) Without limitations on the length of the vectors in A and B,
 44%             C = MULTIPROD(A, B, [DA1 0], [0 DB1]) turns the vectors
 45%             contained in A and B into P×1 and 1×Q matrices, respectively,
 46%             then returns their products, similar to outer products.
 47%             Namely, C = OUTER(A, B, DA1, DB1) is equivalent to
 48%             C = MULTIPROD(CONJ(A), B, [DA1 0], [0 DB1]).
 49%
 50%   Common constraints for all syntaxes:
 51%      The external dimensions of A and B must either be identical or 
 52%      compatible with AX rules. The internal dimensions of each block
 53%      array must be adjacent (DA2 == DA1 + 1 and DB2 == DB1 + 1 are
 54%      required). DA1 and DB1 are allowed to be larger than NDIMS(A) and
 55%      NDIMS(B). In syntaxes 1, 2, and 3, Q == R is required, unless the
 56%      blocks in A or B are scalars. 
 57%
 58%   Array expansion (AX):
 59%      AX is a powerful generalization to N-D of the concept of scalar
 60%      expansion. Indeed, A and B may be scalars, vectors, matrices or
 61%      multi-dimensional arrays. Scalar expansion is the virtual
 62%      replication or annihilation of a scalar which allows you to combine
 63%      it, element by element, with an array X of any size (e.g. X+10,
 64%      X*10, or []-10). Similarly, in MULTIPROD, the purpose of AX is to
 65%      automatically match the size of the external dimensions (EDs) of A
 66%      and B, so that block-by-block products can be performed. ED matching
 67%      is achieved by means of a dimension shift followed by a singleton
 68%      expansion:
 69%      1) Dimension shift (see SHIFTDIM).
 70%            Whenever DA1 ~= DB1, a shift is applied to impose DA1 == DB1.
 71%            If DA1 > DB1, B is shifted to the right by DA1 - DB1 steps.
 72%            If DB1 > DA1, A is shifted to the right by DB1 - DA1 steps.
 73%      2) Singleton expansion (SX).
 74%            Whenever an ED of either A or B is singleton and the
 75%            corresponding ED of the other array is not, the mismatch is
 76%            fixed by virtually replicating the array (or diminishing it to
 77%            length 0) along that dimension.
 78% 
 79%   MULTIPROD is a generalization for N-D arrays of the matrix
 80%   multiplication function MTIMES, with AX enabled. Vector inner, outer,
 81%   and cross products generalized for N-D arrays and with AX enabled are
 82%   performed by DOT2, OUTER, and CROSS2 (MATLAB Central, file #8782).
 83%   Elementwise multiplications (see TIMES) and other elementwise binary
 84%   operations with AX enabled are performed by BAXFUN (MATLAB Central,
 85%   file #23084). Together, these functions make up the “ARRAYLAB toolbox”.
 86%
 87%   Input and output format:
 88%      The size of the EDs of C is determined by AX. Block size is
 89%      determined as follows, for each of the above-listed syntaxes:
 90%      1) C contains P×S matrices along IDs MAX([DA1 DA2], [DB1 DB2]).
 91%      2) Array     Block size     ID(s)
 92%         ----------------------------------------------------
 93%         A         P×Q  (2-D)     [DA1 DA2]
 94%         B         R    (1-D)     DB1
 95%         C (a)     P    (1-D)     MAX(DA1, DB1)
 96%         C (b)     P×Q  (2-D)     MAX([DA1 DA2], [DB1 DB1+1])
 97%         ----------------------------------------------------
 98%         (a) The 1-D blocks in B are not scalars (R > 1).
 99%         (b) The 1-D blocks in B are scalars (R = 1).
100%      3) Array     Block size     ID(s)
101%         ----------------------------------------------------
102%         A           Q  (1-D)     DA1
103%         B         R×S  (2-D)     [DB1 DB2]
104%         C (a)       S  (1-D)     MAX(DA1, DB1)
105%         C (b)     R×S  (2-D)     MAX([DA1 DA1+1], [DB1 DB2])
106%         ----------------------------------------------------
107%         (a) The 1-D blocks in A are not scalars (Q > 1).
108%         (b) The 1-D blocks in A are scalars (Q = 1).
109%      4)     Array     Block size         ID(s)
110%         --------------------------------------------------------------
111%         (a) A         P        (1-D)     DA1
112%             B         Q        (1-D)     DB1
113%             C         MAX(P,Q) (1-D)     MAX(DA1, DB1)
114%         --------------------------------------------------------------
115%         (b) A         P        (1-D)     DA1
116%             B         P        (1-D)     DB1
117%             C         1        (1-D)     MAX(DA1, DB1)
118%         --------------------------------------------------------------
119%         (c) A         P        (1-D)     DA1
120%             B         Q        (1-D)     DB1
121%             C         P×Q      (2-D)     MAX([DA1 DA1+1], [DB1 DB1+1])
122%         --------------------------------------------------------------
123%
124%   Terminological notes:
125%   (*) 1-D and 2-D blocks are generically referred to as "vectors" and 
126%       "matrices", respectively. However, both may be also called
127%       “scalars” if they have a single element. Moreover, matrices with a
128%       single row or column (e.g. 1×3 or 3×1) may be also called “row
129%       vectors” or “column vectors”.
130%   (°) Not to be confused with the "inner dimensions" of the two matrices
131%       involved in a product X * Y, defined as the 2nd dimension of X and
132%       the 1st of Y (DA2 and DB1 in syntaxes 1, 2, 3).
133%
134%   Examples:
135%    1) If  A is .................... a 5×(6×3)×2 array,
136%       and B is .................... a 5×(3×4)×2 array,
137%       C = MULTIPROD(A, B, [2 3]) is a 5×(6×4)×2 array.
138%
139%       A single matrix A pre-multiplies each matrix in B
140%       If  A is ........................... a (1×3)    single matrix,
141%       and B is ........................... a 10×(3×4) 3-D array,
142%       C = MULTIPROD(A, B, [1 2], [3 4]) is a 10×(1×4) 3-D array.
143%
144%       Each matrix in A pre-multiplies each matrix in B (all possible
145%       combinations)
146%       If  A is .................... a (6×3)×5   array,
147%       and B is .................... a (3×4)×1×2 array,
148%       C = MULTIPROD(A, B, [1 2]) is a (6×4)×5×2 array.
149%
150%   2a) If  A is ........................... a 5×(6×3)×2 4-D array,
151%       and B is ........................... a 5×(3)×2   3-D array,
152%       C = MULTIPROD(A, B, [2 3], [2]) is   a 5×(6)×2   3-D array.
153%
154%   2b) If  A is ........................... a 5×(6×3)×2 4-D array,
155%       and B is ........................... a 5×(1)×2   3-D array,
156%       C = MULTIPROD(A, B, [2 3], [2]) is   a 5×(6×3)×2 4-D array.
157%
158%   4a) If both A and B are .................. 5×(6)×2   3-D arrays,
159%       C = MULTIPROD(A, B, 2) is .......... a 5×(1)×2   3-D array, while
160%   4b) C = MULTIPROD(A, B, [2 0], [0 2]) is a 5×(6×6)×2 4-D array
161%
162%   See also DOT2, OUTER, CROSS2, BAXFUN, MULTITRANSP.
163
164% $ Version: 2.1 $
165% CODE      by:            Paolo de Leva
166%                          (Univ. of Rome, Foro Italico, IT)    2009 Jan 24
167%           optimized by:  Paolo de Leva
168%                          Jinhui Bai (Georgetown Univ., D.C.)  2009 Jan 24
169% COMMENTS  by:            Paolo de Leva                        2009 Feb 24
170% OUTPUT    tested by:     Paolo de Leva                        2009 Feb 24
171% -------------------------------------------------------------------------
172
173 narginchk(2, 4) ; % Allow 2 to 4 input arguments
174% error( nargchk(2, 4, nargin) ); % Allow 2 to 4 input arguments
175switch nargin % Setting IDA and/or IDB
176    case 2, idA = [1 2]; idB = [1 2];
177    case 3, idB = idA;
178end
179
180% ESC 1 - Special simple case (both A and B are 2D), solved using C = A * B
181
182     if ndims(a)==2 && ndims(b)==2 && ...
183         isequal(idA,[1 2]) && isequal(idB,[1 2])
184         c = a * b; return
185     end
186
187% MAIN 0 - Checking and evaluating array size, block size, and IDs
188
189     sizeA0 = size(a);
190     sizeB0 = size(b);
191     [sizeA, sizeB, shiftC, delC, sizeisnew, idA, idB, ...
192     squashOK, sxtimesOK, timesOK, mtimesOK, sumOK] = ...
193                                           sizeval(idA,idB, sizeA0,sizeB0);
194
195% MAIN 1 - Applying dimension shift (first step of AX) and 
196%          turning both A and B into arrays of either 1-D or 2-D blocks
197
198     if sizeisnew(1), a = reshape(a, sizeA); end    
199     if sizeisnew(2), b = reshape(b, sizeB); end
200
201% MAIN 2 - Performing products with or without SX (second step of AX)
202
203     if squashOK % SQUASH + MTIMES (fastest engine)
204         c = squash2D_mtimes(a,b, idA,idB, sizeA,sizeB, squashOK); 
205     elseif timesOK % TIMES (preferred w.r. to SX + TIMES)
206         if sumOK, c = sum(a .* b, sumOK);
207         else      c =     a .* b; end
208     elseif sxtimesOK % SX + TIMES
209         if sumOK, c = sum(bsxfun(@times, a, b), sumOK);
210         else      c =     bsxfun(@times, a, b); end
211     elseif mtimesOK % MTIMES (rarely used)
212         c = a * b;
213     end
214
215% MAIN 3 - Reshaping C (by inserting or removing singleton dimensions)
216
217     [sizeC sizeCisnew] = adjustsize(size(c), shiftC, false, delC, false);
218     if sizeCisnew, c = reshape(c, sizeC); end

221function c = squash2D_mtimes(a, b, idA, idB, sizeA, sizeB, squashOK)
222% SQUASH2D_MTIMES  Multiproduct with single-block expansion (SBX).
223%    Actually, no expansion is performed. The multi-block array is
224%    rearranged from N-D to 2-D, then MTIMES is applied, and eventually the
225%    result is rearranged back to N-D. No additional memory is required.
226%    One and only one of the two arrays must be single-block, and its IDs
227%    must be [1 2] (MAIN 1 removes leading singletons). Both arrays
228%    must contain 2-D blocks (MAIN 1 expands 1-D blocks to 2-D).
229
230    if squashOK == 1 % A is multi-block, B is single-block (squashing A)
231
232        % STEP 1 - Moving IDA(2) to last dimension
233        nd = length(sizeA);
234        d2 = idA(2);    
235        order = [1:(d2-1) (d2+1):nd d2]; % Partial shifting
236        a = permute(a, order); % ...×Q
237
238        % STEP 2 - Squashing A from N-D to 2-D  
239        q = sizeB(1);
240        s = sizeB(2);
241        lengthorder = length(order);
242        collapsedsize = sizeA(order(1:lengthorder-1)); 
243        n = prod(collapsedsize);
244        a = reshape(a, [n, q]); % N×Q    
245        fullsize = [collapsedsize s]; % Size to reshape C back to N-D
246
247    else % B is multi-block, A is single-block (squashing B)
248
249        % STEP 1 - Moving IDB(1) to first dimension
250        nd = length(sizeB);
251        d1 = idB(1);    
252        order = [d1 1:(d1-1) (d1+1):nd]; % Partial shifting
253        b = permute(b, order); % Q×...
254
255        % STEP 2 - Squashing B from N-D to 2-D  
256        p = sizeA(1);
257        q = sizeA(2);
258        lengthorder = length(order);
259        collapsedsize = sizeB(order(2:lengthorder)); 
260        n = prod(collapsedsize);
261        b = reshape(b, [q, n]); % Q×N
262        fullsize = [p collapsedsize]; % Size to reshape C back to N-D
263
264    end
265
266    % FINAL STEPS - Multiplication, reshape to N-D, inverse permutation
267    invorder(order) = 1 : lengthorder;
268    c = permute (reshape(a*b, fullsize), invorder);

271function [sizeA, sizeB, shiftC, delC, sizeisnew, idA, idB, ...
272          squashOK, sxtimesOK, timesOK, mtimesOK, sumOK] = ...
273                                          sizeval(idA0,idB0, sizeA0,sizeB0)
274%SIZEVAL   Evaluation of array size, block size, and IDs
275%    Possible values for IDA and IDB:
276%        [DA1 DA2], [DB1 DB2]
277%        [DA1 DA2], [DB1]
278%        [DA1],     [DB1 DB2]
279%        [DA1],     [DB1]
280%        [DA1 0],   [0 DB1]
281%        [0 DA1],   [DB1 0]
282%
283%    sizeA/B     Equal to sizeA0/B0 if RESHAPE is not needed in MAIN 1
284%    shiftC, delC    Variables controlling MAIN 3.
285%    sizeisnew   1x2 logical array; activates reshaping of A and B.
286%    idA/B       May change only if squashOK ~= 0
287%    squashOK    If only A or B is a multi-block array (M-B) and the other
288%                is single-block (1-B), it will be rearranged from N-D to
289%                2-D. If both A and B are 1-B or M-B arrays, squashOK = 0.
290%                If only A (or B) is a M-B array, squashOK = 1 (or 2).
291%    sxtimesOK, timesOK, mtimesOK    Flags controlling MAIN 2 (TRUE/FALSE).
292%    sumOK       Dimension along which SUM is performed. If SUM is not
293%                needed, sumOK = 0.
294
295% Initializing output arguments
296
297    idA = idA0;
298    idB = idB0;
299     squashOK = 0;
300    sxtimesOK = false;
301      timesOK = false;
302     mtimesOK = false;
303        sumOK = 0;
304    shiftC = 0;
305    delC = 0;
306
307% Checking for gross input errors
308
309    NidA = numel(idA);
310    NidB = numel(idB);
311    idA1 = idA(1);
312    idB1 = idB(1);
313    if  NidA>2 || NidB>2 || NidA==0 || NidB==0 || ...
314           ~isreal(idA1) ||    ~isreal(idB1)   || ...
315        ~isnumeric(idA1) || ~isnumeric(idB1)   || ...
316                 0>idA1  ||          0>idB1    || ... % negative 
317         idA1~=fix(idA1) ||  idB1~=fix(idB1)   || ... % non-integer
318         ~isfinite(idA1) ||  ~isfinite(idB1) % Inf or NaN               
319        error('MULTIPROD:InvalidDimensionArgument', ...
320        ['Internal-dimension arguments (e.g., [IDA1 IDA2]) must\n', ...
321         'contain only one or two non-negative finite integers']);
322    end
323
324% Checking Syntaxes containing zeros (4b/c)
325
326    declared_outer = false;
327    idA2 = idA(NidA); % It may be IDA1 = IDA2 (1-D block)
328    idB2 = idB(NidB);
329
330    if any(idA==0) || any(idB==0)
331        
332        % "Inner products": C = MULTIPROD(A, B, [0 DA1], [DB1 0])
333        if idA1==0 && idA2>0 && idB1>0 && idB2==0
334            idA1 = idA2;
335            idB2 = idB1;
336        % "Outer products": C = MULTIPROD(A, B, [DA1 0], [0 DB1]) 
337        elseif idA1>0 && idA2==0 && idB1==0 && idB2>0
338            declared_outer = true;
339            idA2 = idA1;
340            idB1 = idB2;
341        else
342            error('MULTIPROD:InvalidDimensionArgument', ...
343            ['Misused zeros in the internal-dimension arguments\n', ...
344            '(see help heads 4b and 4c)']);
345        end
346        NidA = 1; 
347        NidB = 1;
348        idA = idA1;
349        idB = idB1;
350
351    elseif (NidA==2 && idA2~=idA1+1) || ...  % Non-adjacent IDs
352           (NidB==2 && idB2~=idB1+1)
353        error('MULTIPROD:InvalidDimensionArgument', ...
354        ['If an array contains 2-D blocks, its two internal dimensions', ... 
355        'must be adjacent (e.g. IDA2 == IDA1+1)']);
356    end
357
358% ESC - Case for which no reshaping is needed (both A and B are scalars)
359
360    scalarA = isequal(sizeA0, [1 1]);
361    scalarB = isequal(sizeB0, [1 1]);
362    if scalarA && scalarB
363        sizeA = sizeA0;
364        sizeB = sizeB0;
365        sizeisnew = [false false];
366        timesOK = true; return
367    end
368
369% Computing and checking adjusted sizes
370% The lengths of ADJSIZEA and ADJSIZEB must be >= IDA(END) and IDB(END)
371
372    NsA = idA2 - length(sizeA0); % Number of added trailing singletons
373    NsB = idB2 - length(sizeB0);
374    adjsizeA = [sizeA0 ones(1,NsA)];
375    adjsizeB = [sizeB0 ones(1,NsB)];
376    extsizeA = adjsizeA([1:idA1-1, idA2+1:end]); % Size of EDs
377    extsizeB = adjsizeB([1:idB1-1, idB2+1:end]);
378    p = adjsizeA(idA1);
379    q = adjsizeA(idA2);
380    r = adjsizeB(idB1);
381    s = adjsizeB(idB2);    
382    scalarsinA = (p==1 && q==1);
383    scalarsinB = (r==1 && s==1);
384    singleA = all(extsizeA==1);
385    singleB = all(extsizeB==1);
386    if q~=r && ~scalarsinA && ~scalarsinB && ~declared_outer
387       error('MULTIPROD:InnerDimensionsMismatch', ...
388             'Inner matrix dimensions must agree.');
389    end
390
391% STEP 1/3 - DIMENSION SHIFTING (FIRST STEP OF AX)
392%   Pipeline 1 (using TIMES) never needs left, and may need right shifting.
393%   Pipeline 2 (using MTIMES) may need left shifting of A and right of B.
394
395    shiftA = 0;
396    shiftB = 0;
397    diffBA = idB1 - idA1;    
398    if scalarA % Do nothing
399    elseif singleA && ~scalarsinB, shiftA = -idA1 + 1; %  Left shifting A
400    elseif idB1 > idA1,            shiftA = diffBA;    % Right shifting A        
401    end    
402    if scalarB % Do nothing
403    elseif singleB && ~scalarsinA, shiftB = -idB1 + 1; %  Left shifting B
404    elseif idA1 > idB1,            shiftB = -diffBA;   % Right shifting B
405    end
406
407% STEP 2/3 - SELECTION OF PROPER ENGINE AND BLOCK SIZE ADJUSTMENTS
408
409    addA  = 0; addB  = 0;
410    delA  = 0; delB  = 0;
411    swapA = 0; swapB = 0;
412    idC1 = max(idA1, idB1);
413    idC2 = idC1 + 1;
414    checktimes = false;
415
416    if (singleA||singleB) &&~scalarsinA &&~scalarsinB % Engine using MTIMES
417
418        if singleA && singleB 
419            mtimesOK = true;
420            shiftC=idC1-1; % Right shifting C
421            idC1=1; idC2=2;
422        elseif singleA
423            squashOK = 2;
424            idB = [idB1, idB1+1] + shiftB;
425        else % singleB
426            squashOK = 1;
427            idA = [idA1, idA1+1] + shiftA;
428        end
429
430        if NidA==2 && NidB==2 % 1) 2-D BLOCKS BY 2-D BLOCKS
431            % OK 
432        elseif NidA==2        % 2) 2-D BLOCKS BY 1-D BLOCKS
433            addB=idB1+1; delC=idC2;
434        elseif NidB==2        % 3) 1-D BLOCKS BY 2-D BLOCKS
435            addA=idA1; delC=idC1;
436        else                  % 4) 1-D BLOCKS BY 1-D BLOCKS
437            if declared_outer
438                addA=idA1+1; addB=idB1;
439            else
440                addA=idA1; addB=idB1+1; delC=idC2;
441            end
442        end    
443
444    else % Engine using TIMES (also used if SCALARA || SCALARB)
445        
446        sxtimesOK = true;
447
448        if NidA==2 && NidB==2 % 1) 2-D BLOCKS BY 2-D BLOCKS
449
450            if scalarA || scalarB
451                timesOK=true;                
452            elseif scalarsinA && scalarsinB % scal-by-scal
453                checktimes=true;
454            elseif scalarsinA || scalarsinB || ... % scal-by-mat
455                (q==1 && r==1)  % vec-by-vec ("outer")
456            elseif p==1 && s==1 % vec-by-vec ("inner")
457                swapA=idA1; sumOK=idC1; checktimes=true;
458            elseif s==1 % mat-by-vec
459                swapB=idB1; sumOK=idC2;
460            elseif p==1 % vec-by-mat
461                swapA=idA1; sumOK=idC1;
462            else % mat-by-mat
463                addA=idA2+1; addB=idB1; sumOK=idC2; delC=idC2;
464            end
465
466        elseif NidA==2 % 2) 2-D BLOCKS BY 1-D BLOCKS
467
468            if scalarA || scalarB
469                timesOK=true;                
470            elseif scalarsinA && scalarsinB % scal-by-scal
471                addB=idB1; checktimes=true;
472            elseif scalarsinA % scal-by-vec
473                delA=idA1;
474            elseif scalarsinB % mat-by-scal
475                addB=idB1;
476            elseif p==1 % vec-by-vec ("inner")
477                delA=idA1; sumOK=idC1; checktimes=true;
478            else % mat-by-vec
479                addB=idB1; sumOK=idC2; delC=idC2;
480            end
481
482        elseif NidB==2 % 3) 1-D BLOCKS BY 2-D BLOCKS
483
484            if scalarA || scalarB
485                timesOK=true;                
486            elseif scalarsinA && scalarsinB % scal-by-scal
487                addA=idA1+1; checktimes=true;
488            elseif scalarsinB % vec-by-scal
489                delB=idB2;
490            elseif scalarsinA % scal-by-mat
491                addA=idA1+1;
492            elseif s==1 % vec-by-vec ("inner")
493                delB=idB2; sumOK=idC1; checktimes=true;
494            else % vec-by-mat
495                addA=idA1+1; sumOK=idC1; delC=idC1;
496            end
497
498        else % 4) 1-D BLOCKS BY 1-D BLOCKS
499
500            if scalarA || scalarB
501                timesOK=true;                
502            elseif declared_outer % vec-by-vec ("outer")
503                addA=idA1+1; addB=idB1;
504            elseif scalarsinA && scalarsinB % scal-by-scal
505                checktimes=true;
506            elseif scalarsinA || scalarsinB % vec-by-scal
507            else % vec-by-vec
508                sumOK=idC1; checktimes=true;
509            end
510        end
511    end
512
513% STEP 3/3 - Adjusting the size of A and B. The size of C is adjusted
514%            later, because it is not known yet.
515
516    [sizeA, sizeisnew(1)] = adjustsize(sizeA0, shiftA, addA, delA, swapA);
517    [sizeB, sizeisnew(2)] = adjustsize(sizeB0, shiftB, addB, delB, swapB);
518
519    if checktimes % Faster than calling BBXFUN
520        diff = length(sizeB) - length(sizeA);
521        if isequal([sizeA ones(1,diff)], [sizeB ones(1,-diff)])
522            timesOK = true;
523        end
524    end

527function [sizeA, sizeisnew] = adjustsize(sizeA0, shiftA, addA, delA, swapA)
528% ADJUSTSIZE  Adjusting size of a block array.
529
530    % Dimension shifting (by adding or deleting trailing singleton dim.)
531    if     shiftA>0, [sizeA,newA1] = addsing(sizeA0, 1, shiftA);
532    elseif shiftA<0, [sizeA,newA1] = delsing(sizeA0, 1,-shiftA); 
533    else   sizeA = sizeA0;  newA1  = false;
534    end
535    % Modifying block size (by adding, deleting, or moving singleton dim.)
536    if      addA, [sizeA,newA2] = addsing(sizeA, addA+shiftA, 1); % 1D-->2D 
537    elseif  delA, [sizeA,newA2] = delsing(sizeA, delA+shiftA, 1); % 2D-->1D
538    elseif swapA, [sizeA,newA2] = swapdim(sizeA,swapA+shiftA); % ID Swapping
539    else                 newA2  = false;
540    end
541    sizeisnew = newA1 || newA2;

544function [newsize, flag] = addsing(size0, dim, ns)
545%ADDSING   Adding NS singleton dimensions to the size of an array.
546%   Warning: NS is assumed to be a positive integer.
547%   Example: If the size of A is ..... SIZE0 = [5 9 3]
548%            NEWSIZE = ADDSING(SIZE0, 3, 2) is [5 9 1 1 3]
549
550    if dim > length(size0)
551        newsize = size0;
552        flag = false;
553    else 
554        newsize = [size0(1:dim-1), ones(1,ns), size0(dim:end)];
555        flag = true;
556    end
557
558
559function [newsize, flag] = delsing(size0, dim, ns)
560%DELSING   Removing NS singleton dimensions from the size of an array.
561%   Warning: Trailing singletons are not removed
562%   Example: If the size of A is SIZE0 = [1 1 1 5 9 3]
563%            NEWSIZE = DELSING(SIZE, 1, 3) is  [5 9 3]
564
565    if dim > length(size0)-ns % Trailing singletons are not removed
566        newsize = size0;
567        flag = false;
568    else % Trailing singl. added, so NEWSIZE is guaranteed to be 2D or more
569        newsize = size0([1:dim-1, dim+ns:end, dim]);
570        flag = true;
571    end

574function [newsize, flag] = swapdim(size0, dim)
575%SWAPDIM   Swapping two adjacent dimensions of an array (DIM and DIM+1).
576%   Used only when both A and B are multi-block arrays with 2-D blocks.
577%   Example: If the size of A is .......... 5×(6×3)
578%            NEWSIZE = SWAPIDS(SIZE0, 2) is 5×(3×6)
579
580    newsize = [size0 1]; % Guarantees that dimension DIM+1 exists.
581    newsize = newsize([1:dim-1, dim+1, dim, dim+2:end]);
582    flag = true;
