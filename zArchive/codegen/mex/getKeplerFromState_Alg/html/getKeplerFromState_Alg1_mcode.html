1function [sma, ecc, inc, raan, arg, tru] = getKeplerFromState_Alg(rVect,vVect,gmu)
 2%UNTITLED5 Summary of this function goes here
 3%   Detailed explanation goes here
 4    k_hat = [0; 0; 1];
 5
 6    hVect=crossARH(rVect,vVect);
 7    h=norm(hVect);
 8
 9    r=norm(rVect);
10    v=norm(vVect);
11
12    nVect = crossARH(k_hat, hVect);
13    eVect = ((v^2 - gmu/r)*rVect - dotARH(rVect,vVect)*vVect)/gmu;
14    ecc = norm(eVect);
15
16    Energy=v^2/2 - gmu/r;
17
18    if(ecc ~= 1.0)
19        sma=-gmu/(2*Energy);
20        p = sma*(1-ecc^2);
21    else
22        sma = Inf;
23        p = h^2/gmu;
24    end
25
26    cosInc = hVect(3) / h;
27    inc = real(acos(complex(cosInc)));
28
29    cosRAAN = nVect(1)/norm(nVect);
30    raan = real(acos(complex(cosRAAN)));
31    if(nVect(2) < 0)
32        raan = 2*pi - raan;
33    end
34
35    cosArg = dotARH(nVect, eVect)/(norm(nVect) * norm(eVect));
36    arg = real(acos(complex(cosArg)));
37    if(eVect(3) < 0)
38        arg = 2*pi - arg;
39    end
40
41    cosTru = dotARH(eVect, rVect) / (norm(eVect) * norm(rVect));
42    tru = real(acos(complex(cosTru)));
43    if(dotARH(rVect,vVect)<0)
44        tru = -tru;
45    end
46
47    %%%%%%%%%%
48    % Special Case: Elliptical Equitorial
49    %%%%%%%%%%
50    if(ecc >= 1E-10 && (inc < 1E-10 || abs(inc-pi) < 1E-10))
51        cosLongPeri = eVect(1)/norm(eVect);
52        if(eVect(2) < 0)
53            cosLongPeri = 2*pi - cosLongPeri;
54        end  
55        raan = 0; %these two lines are my convention, hopefully they work
56        arg = cosLongPeri;
57    end
58
59    %%%%%%%%%%
60    % Special Case: Circular Inclined
61    %%%%%%%%%%
62    if(ecc < 1E-10 && inc >= 1E-10 && abs(inc-pi) >= 1E-10)
63        u = real(acos(complex(dotARH(nVect,rVect)/(norm(nVect)*norm(rVect)))));
64        if(rVect(3) < 0)
65            u = 2*pi - u;
66        end
67        tru = u;
68
69        arg = 0;
70    end
71
72    %%%%%%%%%%
73    % Special Case: Circular Equitorial
74    %%%%%%%%%%
75    if(ecc < 1E-10 && (inc < 1E-10 || abs(inc-pi) < 1E-10))
76        l = real(acos(complex(rVect(1)/norm(rVect))));
77        if(rVect(2)<0)
78            l = 2*pi-l;
79        end
80        tru = l;
81
82        raan = 0;
83        arg = 0;
84    end
85end
